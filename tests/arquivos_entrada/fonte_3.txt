def soma_vetor(int dados, int tamanho) {
    int i;
    int soma;
    soma = 0;
    i = 0;
    for(i = 0; i < tamanho; i = i + 1) {
        soma = soma + dados[i];
    }
    return soma;
}

def preenche(int dados, int tamanho) {
    int i;
    i = 0;
    for(i = 0; i < tamanho; i = i + 1) {
        dados[i] = i * 2;
    }
    return tamanho;
}

def ajusta(int valor) {
    int r;
    r = valor;
    if(r > 100) {
        r = r - 50;
    } else {
        r = r + 25;
    }
    return r;
}
