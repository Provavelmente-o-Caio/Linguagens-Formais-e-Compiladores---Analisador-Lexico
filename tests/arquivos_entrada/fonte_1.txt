def acumula(int limite) {
    int i;
    int soma;
    soma = 0;
    i = 0;
    for(i = 0; i < limite; i = i + 1) {
        soma = soma + i;
    }
    return soma;
}

def duplica(int valor) {
    int r;
    r = valor + valor;
    return r;
}
